CREATE OR REPLACE PROCEDURE AUMENTAR (p_cod_producto IN PRODUCTOS.COD_PRODUCTO%TYPE) 
IS
    v_existe        NUMBER;
    v_ventas_total NUMBER;
    v_stock_actual NUMBER;
    v_aumento      NUMBER;
    v_nuevo_stock  NUMBER;
    
    e_producto_no_existe EXCEPTION;
BEGIN
 
   
    SELECT COUNT(*)
    INTO v_existe
    FROM PRODUCTOS
    WHERE COD_PRODUCTO = p_cod_producto;
    
    IF v_existe = 0 THEN
        RAISE e_producto_no_existe;
    END IF;

    
    SELECT STOCK
    INTO v_stock_actual
    FROM PRODUCTOS
    WHERE COD_PRODUCTO = p_cod_producto;

    
    SELECT SUM(UNIDADES)
    INTO v_ventas_total
    FROM VENTAS
    WHERE COD_PRODUCTO = p_cod_producto;

    
    IF v_ventas_total IS NULL THEN
        v_ventas_total := 0;
    END IF;

    
    v_aumento := v_ventas_total + (v_ventas_total * 0.25);
    v_nuevo_stock := v_stock_actual + TRUNC(v_aumento);

   
    UPDATE PRODUCTOS
    SET STOCK = v_nuevo_stock
    WHERE COD_PRODUCTO = p_cod_producto;

    
    DBMS_OUTPUT.PUT_LINE('Cod. Producto: ' || p_cod_producto || ' Stock=' || v_nuevo_stock);

EXCEPTION
    WHEN e_producto_no_existe THEN
        DBMS_OUTPUT.PUT_LINE('Error: El producto con código ' || p_cod_producto || ' no existe en la tabla productos.');
END AUMENTAR;
/

SET SERVEROUTPUT ON;

EXEC AUMENTAR(2);

EXEC AUMENTAR(99);
