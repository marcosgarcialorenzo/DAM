create or replace trigger tr_gestionar_ofertas
before insert or update of precio on libros
for each row
declare
  v_existe number;
begin
  
  select count(*) into v_existe from ofertas where codigo = :new.codigo;

  
  if :new.precio > 30 then
    if v_existe > 0 then
     
      delete from ofertas where codigo = :new.codigo;
    end if;
  else
    
    if v_existe = 0 then
     
      insert into ofertas values (:new.codigo, :new.precio);
    else
      
      update ofertas set precio = :new.precio where codigo = :new.codigo;
    end if;
  end if;
end;
/
