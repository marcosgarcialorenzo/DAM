================================================================================
  SOLUCIONES EJERCICIOS MONGODB
================================================================================

  CONCEPTOS CLAVE USADOS EN ESTE DOCUMENTO
  -----------------------------------------
  find({filtro}, {proyección}) -> Busca documentos. El 1er {} es el filtro (qué
      documentos), el 2º es la proyección (qué campos mostrar: 1=sí, 0=no).
  aggregate([...])  -> Pipeline de operaciones encadenadas (como una cadena de
      pasos: filtrar, agrupar, ordenar, etc.).
  $match    -> Filtra documentos dentro de un aggregate (equivale a WHERE en SQL).
  $group    -> Agrupa documentos por un campo y permite calcular totales,
      promedios, máximos, etc. (equivale a GROUP BY en SQL).
  $project  -> Elige qué campos mostrar/ocultar en la salida (equivale a SELECT).
  $sort     -> Ordena resultados. 1 = ascendente, -1 = descendente.
  $limit    -> Limita el número de resultados devueltos.
  $skip     -> Salta N documentos (útil para paginación).
  $lookup   -> Une dos colecciones por un campo común (equivale a JOIN en SQL).
  $unwind   -> Descompone un array: crea un documento por cada elemento del array.
  $size     -> Devuelve el número de elementos de un array.
  $push     -> Acumula valores en un array dentro de un $group.
  $addToSet -> Como $push pero sin duplicados.
  $sum      -> Suma valores (con $sum: 1 cuenta documentos).
  $avg      -> Calcula la media de un campo numérico.
  $max/$min -> Valor máximo/mínimo de un campo.
  /^TEXTO/  -> Expresión regular: busca documentos cuyo campo empiece por TEXTO.
  $gt/$lt   -> Comparadores: $gt = mayor que, $lt = menor que.
  $or       -> Condición OR entre varios filtros.
  $set      -> Modifica el valor de un campo en un updateOne/updateMany.
  $addToSet (update) -> Añade un valor a un array solo si no existe ya.
  $pull     -> Elimina elementos de un array que cumplan una condición.
  countDocuments() -> Cuenta cuántos documentos cumplen el filtro.
  insertOne/insertMany -> Inserta uno o varios documentos.
  updateOne -> Modifica el primer documento que cumple el filtro.
  deleteOne -> Elimina el primer documento que cumple el filtro.
  findOne() -> Devuelve el primer documento de la colección.


================================================================================
  9.1 - COLECCIÓN MULTINACIONAL
================================================================================

-- 1. Mostrar el nombre de la empresa, del jefe y del empleado de toda la colección
--    find({}) sin filtro devuelve todos los documentos.
--    La proyección { _id:0, campo:1 } indica qué campos mostrar (y oculta _id).
db.multinacional.find({}, { _id: 0, nombreEmpresa: 1, nombreJefe: 1, nombreEmpleado: 1 })

-- 2. Mostrar todos los datos que tengan la categoría 'ANALISTA'
--    Filtro simple por igualdad: solo devuelve los documentos donde categoria="ANALISTA".
db.multinacional.find({ categoria: "ANALISTA" })

-- 3. Mostrar el nombre del jefe y empleado de la empresa GETRONICS
--    Combina filtro (empresa=GETRONICS) con proyección (solo jefe y empleado).
db.multinacional.find(
  { nombreEmpresa: "GETRONICS" },
  { _id: 0, nombreJefe: 1, nombreEmpleado: 1 }
)

-- 4. Mostrar jefe y empleado de todos los ANALISTAS agrupados por nombre de empresa
--    Filtra por ANALISTA y ordena por empresa (sort ascendente = 1).
--    "Agrupados" aquí significa ordenados, no un $group real.
db.multinacional.find(
  { categoria: "ANALISTA" },
  { _id: 0, nombreEmpresa: 1, nombreJefe: 1, nombreEmpleado: 1 }
).sort({ nombreEmpresa: 1 })

-- 5. Mostrar jefe y empleado de ANALISTAS ordenados por nombre del jefe
--    Igual que el anterior pero ordenado por el nombre del jefe.
db.multinacional.find(
  { categoria: "ANALISTA" },
  { _id: 0, nombreJefe: 1, nombreEmpleado: 1 }
).sort({ nombreJefe: 1 })

-- 6. Mostrar jefe y empleado de quienes ganan menos de 1800 euros
--    $lt = "less than" (menor que). Filtra sueldoNeto < 1800.
db.multinacional.find(
  { sueldoNeto: { $lt: 1800 } },
  { _id: 0, nombreJefe: 1, nombreEmpleado: 1 }
)

-- 7. Jefe y empleado cuya fecha de ingreso fue a partir del año 2002
--    $gt = "greater than" (mayor que). Compara el string de fecha (funciona porque
--    el formato "AAAA/MM/DD" se ordena lexicográficamente igual que cronológicamente).
db.multinacional.find(
  { fechaIngreso: { $gt: "2002" } },
  { _id: 0, nombreJefe: 1, nombreEmpleado: 1 }
)

-- 8. Contar cuántos empleados tienen la categoría "TECNICO"
--    countDocuments devuelve solo el número, no los documentos.
db.multinacional.countDocuments({ categoria: "TECNICO" })

-- 9. Mostrar cuántos empleados hay por cada empresa (AGGREGATE)
--    $group agrupa todos los documentos por nombreEmpresa.
--    $sum: 1 suma 1 por cada documento del grupo, obteniendo el total.
db.multinacional.aggregate([
  { $group: { _id: "$nombreEmpresa", totalEmpleados: { $sum: 1 } } }
])

-- 10. Obtener el sueldo promedio por categoría
--     $avg calcula la media del campo sueldoNeto dentro de cada grupo (categoría).
db.multinacional.aggregate([
  { $group: { _id: "$categoria", sueldoPromedio: { $avg: "$sueldoNeto" } } }
])

-- 11. Obtener el sueldo máximo y mínimo por empresa
--     Se pueden calcular varios acumuladores en el mismo $group.
--     $max y $min obtienen el valor más alto y más bajo del campo.
db.multinacional.aggregate([
  {
    $group: {
      _id: "$nombreEmpresa",
      sueldoMaximo: { $max: "$sueldoNeto" },
      sueldoMinimo: { $min: "$sueldoNeto" }
    }
  }
])

-- 12. Contar cuántos empleados hay por jefe
--     Agrupa por nombreJefe y cuenta los documentos de cada grupo.
db.multinacional.aggregate([
  { $group: { _id: "$nombreJefe", empleadosACargo: { $sum: 1 } } }
])

-- 13. Obtener edad promedio de los empleados por categoría
--     $avg sobre el campo edad, agrupando por categoría.
db.multinacional.aggregate([
  { $group: { _id: "$categoria", edadPromedio: { $avg: "$edad" } } }
])

-- 14. Mostrar cuántos empleados hay por categoría ordenados de mayor a menor
--     Tras el $group, $sort ordena por el campo calculado "total" de forma
--     descendente (-1). Es como un GROUP BY + ORDER BY DESC en SQL.
db.multinacional.aggregate([
  { $group: { _id: "$categoria", total: { $sum: 1 } } },
  { $sort: { total: -1 } }
])

-- 15. Mostrar nombres de empleados y su empresa para los que ganan más de 2000 (MATCH)
--     $match filtra antes de procesar (más eficiente que filtrar al final).
--     $project selecciona qué campos incluir en la salida.
db.multinacional.aggregate([
  { $match: { sueldoNeto: { $gt: 2000 } } },
  { $project: { _id: 0, nombreEmpresa: 1, nombreEmpleado: 1, sueldoNeto: 1 } }
])

-- 16. Mostrar cuántos empleados hay en cada combinación de empresa y categoría
--     El _id del $group puede ser un objeto con varios campos para agrupar
--     por múltiples criterios a la vez (equivale a GROUP BY empresa, categoria).
db.multinacional.aggregate([
  {
    $group: {
      _id: { empresa: "$nombreEmpresa", categoria: "$categoria" },
      total: { $sum: 1 }
    }
  }
])

-- 17. Mostrar la lista de empleados más jóvenes por categoría (edad mínima)
--     $min obtiene el valor mínimo del campo edad dentro de cada grupo.
db.multinacional.aggregate([
  { $group: { _id: "$categoria", edadMinima: { $min: "$edad" } } }
])

-- 18. Mostrar el sueldo total por empresa (suma de sueldos netos)
--     $sum sobre un campo numérico (sueldoNeto) suma todos los valores del grupo.
--     Distinto de $sum:1 que solo cuenta documentos.
db.multinacional.aggregate([
  { $group: { _id: "$nombreEmpresa", sueldoTotal: { $sum: "$sueldoNeto" } } }
])

-- 19. Mostrar las 5 empresas con mayor sueldo promedio (LIMIT)
--     Pipeline de 3 pasos: agrupar + calcular promedio -> ordenar desc -> limitar a 5.
db.multinacional.aggregate([
  { $group: { _id: "$nombreEmpresa", promedioSueldo: { $avg: "$sueldoNeto" } } },
  { $sort: { promedioSueldo: -1 } },
  { $limit: 5 }
])

-- 20. Mostrar solo la tercera de las 5 anteriores (SKIP)
--     Mismo pipeline que el anterior, pero $skip:2 salta los 2 primeros resultados
--     y $limit:1 devuelve solo 1, obteniendo así el tercero.
db.multinacional.aggregate([
  { $group: { _id: "$nombreEmpresa", promedioSueldo: { $avg: "$sueldoNeto" } } },
  { $sort: { promedioSueldo: -1 } },
  { $limit: 5 },
  { $skip: 2 },
  { $limit: 1 }
])


================================================================================
  9.2 - COLECCIÓN PERSONAS
================================================================================

-- 1. Mostrar todos los datos de la colección persona
--    find() sin argumentos devuelve todos los documentos con todos sus campos.
db.persona.find()

-- 2. Personas mayores de 18 años
--    $gt: 18 significa "mayor que 18" (no incluye los de exactamente 18).
db.persona.find({ Edad: { $gt: 18 } })

-- 3. Mostrar id, nombre y apellidos de las 'Marisas' encontradas
--    _id: 1 lo incluye explícitamente (por defecto siempre aparece salvo que lo ocultes).
db.persona.find({ Nombre: "Marisa" }, { _id: 1, Nombre: 1, Apellidos: 1 })

-- 4. Mostrar nombre y apellidos de las 'Marisas' encontradas
--    _id: 0 oculta el _id. Solo se muestran Nombre y Apellidos.
db.persona.find({ Nombre: "Marisa" }, { _id: 0, Nombre: 1, Apellidos: 1 })

-- 5. Mostrar nombre y apellidos de los menores de 20 años
--    $lt: 20 significa "menor que 20" (no incluye los de exactamente 20).
db.persona.find({ Edad: { $lt: 20 } }, { _id: 0, Nombre: 1, Apellidos: 1 })

-- 6. Mostrar las personas que se llamen "Marisa" y que sean mayores de 17 años
--    Dos condiciones en el mismo objeto = AND implícito (ambas deben cumplirse).
db.persona.find({ Nombre: "Marisa", Edad: { $gt: 17 } })

-- 7. Mostrar la persona que se llamen "Marisa" o que tengan más de 20 años
--    $or recibe un array de condiciones: se devuelven documentos que cumplan
--    al menos una de ellas.
db.persona.find({ $or: [ { Nombre: "Marisa" }, { Edad: { $gt: 20 } } ] })

-- 8. Ordenar descendentemente por el nombre
--    sort({campo: -1}) ordena de Z a A (descendente).
db.persona.find().sort({ Nombre: -1 })

-- 9. Ordenar ascendentemente por el nombre
--    sort({campo: 1}) ordena de A a Z (ascendente).
db.persona.find().sort({ Nombre: 1 })

-- 10. Contar cuántos amigos tiene cada persona
--     $project con $size calcula el tamaño del array Amigos de cada documento.
--     No agrupa, simplemente añade un campo calculado a cada documento.
db.persona.aggregate([
  {
    $project: {
      _id: 1,
      Nombre: 1,
      Cantidadpersona: { $size: "$Amigos" }
    }
  }
])

-- 11. Mostrar solo el primer registro de la colección persona
--     findOne() devuelve directamente el primer documento (sin array).
--     La alternativa con limit(1) devuelve un cursor con un solo documento.
db.persona.findOne()
-- o también:
db.persona.find().limit(1)

-- 12. Mostrar solo el último registro de la colección persona
--     Se ordena por _id de forma descendente (-1) para que el último quede primero,
--     y luego limit(1) coge solo ese.
db.persona.find().sort({ _id: -1 }).limit(1)

-- 13. Mostrar las personas que tengan 18 años ordenando sus amigos descendentemente
--     $match filtra los de 18 años. $sortArray (disponible desde MongoDB 5.2)
--     reordena el array Amigos dentro de cada documento por Nombre descendente.
db.persona.aggregate([
  { $match: { Edad: 18 } },
  {
    $project: {
      _id: 1,
      Nombre: 1,
      Apellidos: 1,
      Edad: 1,
      Aficiones: 1,
      Amigos: {
        $sortArray: { input: "$Amigos", sortBy: { Nombre: -1 } }
      }
    }
  }
])

-- 14. Contar cuántas personas tienen cada afición
--     $unwind "desdobla" el array Aficiones: crea un documento por cada afición.
--     Luego $group agrupa esos documentos por afición y cuenta cuántos hay.
db.persona.aggregate([
  { $unwind: "$Aficiones" },
  { $group: { _id: "$Aficiones", Cantidad: { $sum: 1 } } }
])

-- 15. Mostrar el nombre de los amigos de todas las personas de la colección
--     $unwind sobre Amigos genera un documento por cada amigo de cada persona.
--     $project muestra el _id de la persona y el Nombre del amigo.
db.persona.aggregate([
  { $unwind: "$Amigos" },
  { $project: { _id: 1, Nombre: "$Amigos.Nombre" } }
])

-- 16. Insertar otro registro más
--     insertOne añade un nuevo documento a la colección con los campos indicados.
db.persona.insertOne({
  "_id": 16,
  "Nombre": "Carlos",
  "Apellidos": "Pérez",
  "Edad": 22,
  "Aficiones": ["baloncesto", "videojuegos"],
  "Amigos": [
    { "Nombre": "Juan", "Edad": 30 },
    { "Nombre": "Ana", "Edad": 25 }
  ]
})
-- Comprobar que se insertó correctamente:
db.persona.find({ _id: 16 })

-- 17. Modificar el Nombre de Carlos (id:16) poniendo José Carlos
--     updateOne busca el documento con _id:16 y $set cambia solo el campo Nombre.
--     El resto de campos del documento no se tocan.
db.persona.updateOne(
  { _id: 16 },
  { $set: { Nombre: "Jose Carlos" } }
)
-- Comprobar:
db.persona.find({ _id: 16 })

-- 18. Añadir a Jose Carlos una nueva afición "futbol" usando $addToSet
--     $addToSet añade el valor al array solo si no existe ya (evita duplicados).
--     Si ya hubiera "futbol" en Aficiones, no lo añadiría de nuevo.
db.persona.updateOne(
  { _id: 16 },
  { $addToSet: { Aficiones: "futbol" } }
)
-- Comprobar:
db.persona.find({ _id: 16 })

-- 19. Eliminar de los amigos de Jose Carlos a Ana (usando $pull)
--     $pull elimina del array Amigos todos los subdocumentos donde Nombre="Ana".
db.persona.updateOne(
  { _id: 16 },
  { $pull: { Amigos: { Nombre: "Ana" } } }
)
-- Comprobar:
db.persona.find({ _id: 16 })

-- 20. Eliminar el registro completo de Jose Carlos (deleteOne)
--     deleteOne elimina el primer documento que cumple el filtro.
--     Como _id es único, elimina exactamente ese documento.
db.persona.deleteOne({ _id: 16 })
-- Comprobar que ya solo hay 15 registros:
db.persona.find().count()

  9.3 - COLECCIÓN CLÍNICA (Hospitales y Médicos)
================================================================================

-- 1. Obtener todos los hospitales y sus médicos
--    $lookup hace un JOIN entre hospitales y medicos.
--    Une hospitales._id con medicos.hospital_id y guarda el resultado en "medicos".
--    Los hospitales sin médicos aparecen con el array "medicos" vacío.
db.hospitales.aggregate([
  {
    $lookup: {
      from: "medicos",             -- colección con la que se une
      localField: "_id",           -- campo de hospitales
      foreignField: "hospital_id", -- campo de medicos que coincide
      as: "medicos"                -- nombre del array resultado
    }
  }
])

-- 2. Contar el número de médicos por hospital
--    Tras el $lookup, $project usa $size para contar cuántos elementos
--    tiene el array "medicos" en cada hospital.
db.hospitales.aggregate([
  {
    $lookup: {
      from: "medicos",
      localField: "_id",
      foreignField: "hospital_id",
      as: "medicos"
    }
  },
  {
    $project: {
      nombre: 1,
      totalMedicos: { $size: "$medicos" }
    }
  }
])

-- 3. Mostrar los hospitales con médicos en Cardiología
--    Tras el $lookup, $match filtra solo los hospitales donde algún médico
--    del array tenga especialidad="Cardiología".
db.hospitales.aggregate([
  {
    $lookup: {
      from: "medicos",
      localField: "_id",
      foreignField: "hospital_id",
      as: "medicos"
    }
  },
  {
    $match: { "medicos.especialidad": "Cardiología" }
  },
  {
    $project: { _id: 1, nombre: 1, ciudad: 1 }
  }
])

-- 4. Obtener nombres y direcciones de hospitales sin médicos
--    Tras el $lookup, filtra los que tienen el array "medicos" vacío ($size: 0).
--    Esto equivale a un LEFT JOIN ... WHERE medicos.id IS NULL en SQL.
db.hospitales.aggregate([
  {
    $lookup: {
      from: "medicos",
      localField: "_id",
      foreignField: "hospital_id",
      as: "medicos"
    }
  },
  {
    $match: { medicos: { $size: 0 } }
  },
  {
    $project: { _id: 0, nombre: 1, direccion: 1 }
  }
])

-- 5. Mostrar los médicos por especialidad
--    $group agrupa por especialidad. $push acumula los nombres en un array.
--    Resultado: un documento por especialidad con la lista de sus médicos.
db.medicos.aggregate([
  {
    $group: {
      _id: "$especialidad",
      medicos: { $push: "$nombre" },
      total: { $sum: 1 }
    }
  }
])

-- 6. Obtener la información de médicos y su hospital
--    $lookup une médicos con hospitales por hospital_id.
--    $unwind convierte el array hospitalInfo (de 1 elemento) en un campo simple.
--    $project construye la salida con campos de ambas colecciones.
db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "hospitalInfo"
    }
  },
  { $unwind: "$hospitalInfo" },
  {
    $project: {
      _id: 0,
      especialidad: 1,
      medico: "$nombre",
      hospital: "$hospitalInfo.nombre",
      ciudad: "$hospitalInfo.ciudad"
    }
  }
])

-- 7. Mostrar los médicos que trabajan en hospitales de Madrid
--    Mismo JOIN que el ejercicio 6, pero con un $match al final para
--    filtrar solo los hospitales cuya ciudad sea "Madrid".
db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "hospitalInfo"
    }
  },
  { $unwind: "$hospitalInfo" },
  { $match: { "hospitalInfo.ciudad": "Madrid" } },
  {
    $project: {
      _id: 0,
      nombre: 1,
      especialidad: 1,
      hospital: { nombre: "$hospitalInfo.nombre" }
    }
  }
])

-- 8. Mostrar los teléfonos de hospitales que tienen médicos en Oncología
--    Primero filtra médicos de Oncología ($match), luego une con hospitales
--    para obtener los teléfonos de esos hospitales.
db.medicos.aggregate([
  { $match: { especialidad: "Oncología" } },
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "hospitalInfo"
    }
  },
  { $unwind: "$hospitalInfo" },
  {
    $project: {
      _id: 0,
      nombre: "$hospitalInfo.nombre",
      telefonos: "$hospitalInfo.telefonos"
    }
  }
])

-- 9. Mostrar los médicos con el nombre del hospital ordenados por ciudad
--    JOIN médicos-hospitales y $sort por ciudad del hospital (ascendente).
db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "hospitalInfo"
    }
  },
  { $unwind: "$hospitalInfo" },
  {
    $project: {
      _id: 1,
      especialidad: 1,
      medico: "$nombre",
      hospital: "$hospitalInfo.nombre",
      ciudad: "$hospitalInfo.ciudad"
    }
  },
  { $sort: { ciudad: 1 } }
])

-- 12. Contar cuántos médicos hay en total por ciudad
--     JOIN médicos-hospitales, luego $group por ciudad y $sum:1 cuenta médicos.
--     $sort descendente para ver primero las ciudades con más médicos.
db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "hospitalInfo"
    }
  },
  { $unwind: "$hospitalInfo" },
  {
    $group: {
      _id: "$hospitalInfo.ciudad",
      total_medicos: { $sum: 1 }
    }
  },
  { $sort: { total_medicos: -1 } }
])

-- 13. Mostrar las especialidades únicas por ciudad (usando $addToSet)
--     $addToSet en el $group acumula especialidades sin repetir.
--     Resultado: un documento por ciudad con un array de sus especialidades únicas.
db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "hospitalInfo"
    }
  },
  { $unwind: "$hospitalInfo" },
  {
    $group: {
      _id: "$hospitalInfo.ciudad",
      especialidades: { $addToSet: "$especialidad" }
    }
  }
])

-- 14. Mostrar todos los hospitales con el número total de especialidades distintas
--     $addToSet acumula especialidades sin duplicados, luego $size cuenta cuántas
--     especialidades distintas tiene cada hospital.
db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "hospitalInfo"
    }
  },
  { $unwind: "$hospitalInfo" },
  {
    $group: {
      _id: "$hospitalInfo.nombre",
      especialidades: { $addToSet: "$especialidad" }
    }
  },
  {
    $project: {
      hospital: "$_id",
      total_especialidades: { $size: "$especialidades" }
    }
  }
])

-- 18. Contar el número de hospitales por ciudad
--     $group sobre la colección hospitales directamente (sin JOIN).
--     Agrupa por ciudad y cuenta cuántos hospitales hay en cada una.
db.hospitales.aggregate([
  {
    $group: {
      _id: "$ciudad",
      total_hospitales: { $sum: 1 }
    }
  },
  { $sort: { total_hospitales: -1 } }
])


================================================================================
  9.4 - COLECCIÓN BANCOS (Clientes, Cuentas y Sucursales)
================================================================================

-- 1. Mostrar todas las cuentas
--    find() sin argumentos devuelve todos los documentos de la colección.
db.cuentas.find()

-- 2. Mostrar todas las sucursales
db.sucursales.find()

-- 3. Mostrar el total de clientes con apellido PEREZ
--    countDocuments con filtro cuenta solo los que cumplen la condición.
db.clientes.countDocuments({ apellidos: "PEREZ" })

-- 4. Mostrar las sucursales cuya dirección empieza por ALCALA
--    /^ALCALA/ es una expresión regular: ^ indica "empieza por".
db.sucursales.find({ direccion: /^ALCALA/ })

-- 5. Mostrar el total de cuentas
--    countDocuments sin filtro cuenta todos los documentos de la colección.
db.cuentas.countDocuments()

-- 6. Mostrar las cuentas de clientes que se apelliden SANCHEZ
--    $match filtra clientes con apellido SANCHEZ.
--    $lookup une con cuentas por codcuenta para obtener los datos de la cuenta.
db.clientes.aggregate([
  { $match: { apellidos: "SANCHEZ" } },
  {
    $lookup: {
      from: "cuentas",
      localField: "codcuenta",
