CREATE OR REPLACE PROCEDURE DATOSADICIONELS (p_descripcion VARCHAR2) AS
    v_stock_actual NUMBER;
    v_remplazo VARCHAR2(50);
    v_ultima VARCHAR2(1);
    v_penultima VARCHAR2(1);
    v_antepenultima VARCHAR2(1);
    v_num_p NUMBER;
BEGIN
   
    SELECT STOCK INTO v_stock_actual
    FROM PRODUCTOS
    WHERE DESCRIPCION = p_descripcion;

    DBMS_OUTPUT.PUT_LINE('-------------------------------------------');
    DBMS_OUTPUT.PUT_LINE('DATOS ADICIONALES');
    DBMS_OUTPUT.PUT_LINE('-------------------------------------------');
    DBMS_OUTPUT.PUT_LINE('Descripción del producto: ' || p_descripcion);

    
    IF v_stock_actual < 5000 THEN
        DBMS_OUTPUT.PUT_LINE('EL PRODUCTO MENOS VENDIDO TIENE POCAS UNIDADES');
    ELSE
        DBMS_OUTPUT.PUT_LINE('EL PRODUCTO MENOS VENDIDO TIENE MUCHAS UNIDADES');
    END IF;

   
    v_remplazo := REPLACE(REPLACE(p_descripcion, 'P', 'B'), 'p', 'b');
    DBMS_OUTPUT.PUT_LINE('Sustituyendo P por B: ' || v_remplazo);

 
    v_ultima        := SUBSTR(p_descripcion, -1, 1);
    v_penultima     := SUBSTR(p_descripcion, -2, 1);
    v_antepenultima := SUBSTR(p_descripcion, -3, 1);

    DBMS_OUTPUT.PUT_LINE('Último carácter: ' || v_ultima);
    DBMS_OUTPUT.PUT_LINE('Penúltimo carácter: ' || v_penultima);
    DBMS_OUTPUT.PUT_LINE('Antepenúltimo carácter: ' || v_antepenultima);

    v_num_p := LENGTH(p_descripcion) - LENGTH(REPLACE(REPLACE(p_descripcion, 'P', ''), 'p', ''));
    DBMS_OUTPUT.PUT_LINE('Cantidad de P/p: ' || v_num_p);
    
END;
/

CREATE OR REPLACE PROCEDURE PRODUCTO_MENOS_VENDIDO AS
    v_cod_prod PRODUCTOS.COD_PRODUCTO%TYPE;
    v_desc PRODUCTOS.DESCRIPCION%TYPE;
    v_nuevo_stock PRODUCTOS.STOCK%TYPE;
BEGIN
    
    SELECT COD_PRODUCTO INTO v_cod_prod
    FROM (
        SELECT COD_PRODUCTO
        FROM VENTAS
        GROUP BY COD_PRODUCTO
        ORDER BY SUM(UNIDADES) ASC
    )
    WHERE ROWNUM = 1;

    
    SELECT DESCRIPCION INTO v_desc 
    FROM PRODUCTOS 
    WHERE COD_PRODUCTO = v_cod_prod;

    
    UPDATE PRODUCTOS
    SET STOCK = STOCK - 5
    WHERE COD_PRODUCTO = v_cod_prod;

    
    SELECT STOCK INTO v_nuevo_stock 
    FROM PRODUCTOS 
    WHERE COD_PRODUCTO = v_cod_prod;
    
    DBMS_OUTPUT.PUT_LINE('El producto menos vendido es: ' || v_cod_prod || ' con descripcion ' || v_desc || ')');

    
    DATOSADICIONELS(v_desc);

END;
/


SET SERVEROUTPUT ON;

EXEC PRODUCTO_MENOS_VENDIDO;
